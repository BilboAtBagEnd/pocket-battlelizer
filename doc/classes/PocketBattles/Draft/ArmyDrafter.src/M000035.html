            def reserve_seed_units
                troop_refs_seeded = (@open_units.inject([]) {|memo, u| memo + u.troops}).map {|t| t.reference}
                @troops.reject! { |t| troop_refs_seeded.include? t.reference }
                @points_left -= sum_points(@open_units)
                @troops.shuffle!
            end
