            def _unit_to_s(u)
                s = "#{u.name} #{u.reference} "
                if u.shooting.size > 0 
                    s += "s#{u.shooting} "
                end
                if u.engagement.size > 0
                    s += "e#{u.engagement} "
                end
                s += "(#{u.points})[#{u.formation}] !#{u.wounds}!"
                if u.powers.size > 0 
                    s += " {#{u.powers.join('|')}}"
                end
                s
            end
