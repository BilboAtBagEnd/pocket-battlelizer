        def to_s
            s = "#{@name} #{self.reference}, "
            s += "s#{@shooting.join('-')}, " unless @shooting.empty?
            s += "e#{@engagement.join('-')}, " unless @engagement.empty?
            s += "(#{@points})[#{@formation}] !#{@wounds}!"
            s += " {#{@powers.join(', ')}}" unless @powers.empty?
            s
        end
